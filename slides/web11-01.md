// libs/itemsService.js
getItems(ids) {
  const uri = `${kintoneApp.base}records.json?app=${kintoneApp.id}`;
  debug(`GET ${uri}`);

  return fetch(uri, {
    agent,
    headers: { 'X-Cybozu-API-Token': kintoneApp.token }
  })
    .then(res => res.json())
    .then(data => {
      debug(data);

      return [{id: 1, name: 'ダミー商品', summary: 'ダミーの商品', price: 1280}]
    }, debug);
}
